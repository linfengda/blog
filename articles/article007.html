二叉排序树

用c语言描述

                  
#pragma once
#include <stdio.h>
#include <stdlib.h>


typedef struct BBTreeNode{
	int data;
	struct BBTreeNode *leftChild;
	struct BBTreeNode *rightChild;
}BBTreeNode;


//中序遍历
void inOrder(BBTreeNode *root, void visit(BBTreeNode *curr)) {
	if (root != NULL) {
		inOrder(root->leftChild, visit);
		visit(root);
		inOrder(root->rightChild, visit);
	}
}

void visit(BBTreeNode *curr) {
	printf("%d, ", curr->data);
}

//查找某个结点
BBTreeNode * search(BBTreeNode *root, int e) {
	if (root != NULL) {
		BBTreeNode *p = root;
		while(p != NULL){
			if(p->data == e) return p;
			if(p->data > e) p = p->leftChild;
			else p = p->rightChild;
		}
	}
	return NULL;
}

//插入结点
int insertNode(BBTreeNode *root, int e) {
	BBTreeNode *parent = NULL;
	BBTreeNode *current = root;
	while(current != NULL){ //向下找到末端插入位置
		parent = current;
		if(current->data == e) return 0; //数据元素已存在
		if(current->data > e) current = current->leftChild;
		else current = current->rightChild;
	}

	BBTreeNode *p = (BBTreeNode*)malloc(sizeof(BBTreeNode));
	p->data = e;
	p->leftChild = NULL;
	p->rightChild = NULL;
	if(root == NULL) root = p; //新结点成为根结点
	else{
		if(parent->data > e) parent->leftChild = p;
		else parent->rightChild = p;
	}
	return 1;
}

//插入结点
int deleteNode(BBTreeNode *root, int e) {
	//1. 要删除的结点没有孩子结点：直接删除
	//2. 要删除的结点只有左孩子结点：删除该结点，并用左孩子结点代替原结点位置
	//3. 要删除的结点只有右孩子结点：删除该结点，并用右孩子结点代替原结点位置
	//4. 要删除的结点有左右孩子结点：删除该节点，并将左右孩子结点重新调整为顺序二叉树，并用根结点代替原结点位置
	return -1;
}

#include <stdio.h>
#include <stdlib.h>

#include "BBTreeNode.h"


void main(void) {

	int test[] = {4,5,7,2,1,9,8,11,3};

	BBTreeNode *root = NULL;
	for(int i = 0; i < 9; i++){
		if(i == 0){
			root = (BBTreeNode*)malloc(sizeof(BBTreeNode));
			root->data = test[0];
			root->leftChild = NULL;
			root->rightChild = NULL;
		}
		else insertNode(root, test[i]);
	}
	printf("\n中序遍历");
	inOrder(root, visit);

	BBTreeNode *find = search(root, 8);
	if(find == NULL) printf("\n元素不存在");
	else  printf("\n元素%d存在", find->data);
}

二叉排序树性能分析

可以看到最坏情况下平均查找长度为O(n)，一般情况下平均查找长度为lb(n)
为了防止最坏情况发生，可以将二叉排序树改造成二叉平衡树，所谓二叉平衡树，就是左右子树深度之差不超过1的二叉排序树。
